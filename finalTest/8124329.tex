\item[{\bf Q3.}] 
\begin{itemize}
\item A convolução das transformadas de Fourier de duas
  funções é a operação definida por:
$$F(\omega)\otimes G(\omega) = \int\limits_{k=-\infty}^{\infty} F(\nu)G(\omega- \nu) d\nu$$
Mostre que a convolução de $F(\omega)$ e $G(\omega)$ é 
igual a transformada de Fourier de $f(x)$ e $g(x)$.

  \vspace{0.5cm}
  \textbf{Resolução:} \\
  Temos $\mathcal{F}[ f(x)g(x) ] = \int\limits_{-\infty}^{\infty} f(x)g(x)e^{-i \omega x} dx $
  $$= \int\limits_{-\infty}^{\infty} f(x) \left [ \frac{1}{2\pi} \int\limits_{-\infty}^{\infty} F(\nu)e^{i \nu x} d\nu \right ]e^{-i \omega x} dx$$
  $$= \frac{1}{2\pi} \int\limits_{-\infty}^{\infty} F(\nu)\left [ \int\limits_{-\infty}^{\infty} f(x)e^{-ix( \omega - \nu )} dx \right ]d\nu$$
  tal que $\int\limits_{-\infty}^{\infty} f(x)e^{-ix( \omega - \nu )} dx = G(\omega - \nu)$. Assim,
  $$= \frac{1}{2\pi} \int\limits_{-\infty}^{\infty} F(\nu) G( \omega - \nu )d\nu$$
  Logo, $$ \int\limits_{-\infty}^{\infty} F(\nu) G( \omega - \nu )d\nu = F(\omega) \otimes  G(\omega)$$

  \vspace{0.5cm}

\newpage
\item Calcule, ``na mão'' (explicitando os cálculos como fizemos em
  classe) a transformada de Fourier da imagem abaixo:
\begin{center}
$
\begin{bmatrix}
    1 & -1 \\
    2 & 0  \\
\end{bmatrix}
$
\end{center}
\end{itemize}
  
  \vspace{0.5cm}
  \textbf{Resolução:} \\
  Tendo $F(k) = \sum_{n=0}^{N-1} f(n) e{^{-j2\pi \frac{k}{N}n}}$ e N = 4, teremos: 
  \\
  \vspace{0.5cm}
  \subitem $\cdot F(1) = \sum_{n=0}^{3} f(n) e{^{-j2\pi \frac{1}{4}n}} $
  $$= (1)e^{0} + (-1)(e^{-j\frac{\pi}{2}})+(2)(e^{-j\pi}) + 0= 1 +(-1)(-j)+(2)(-1) = -1+j $$ \\
  \\
  \subitem $\cdot F(-1) = \sum_{n=0}^{3} f(n) e{^{-j2\pi (-\frac{1}{4})n}} $
  $$= (1)e^{0} + (-1)(e^{j\frac{\pi}{2}})+(2)(e^{j\pi}) + 0 = 1 +(-1)(j)+(2)(-1) = -1-j $$ 
  \\
  \subitem $\cdot F(2) = \sum_{n=0}^{3} f(n) e{^{-j2\pi \frac{2}{4}n}} = \sum_{n=0}^{3} f(n) e{^{-j\pi n}} $
  $$= (1)e^{0} + (-1)(e^{-j\pi}) + (2)(e^{-j\pi 2}) + 0 = 1 +(-1)(-1)+(2)(1) = 1 + 1 + 2 = 4 + 0j $$
  \\
  \subitem $\cdot F(0) = \sum_{n=0}^{3} f(n) e{^{-j2\pi \frac{0}{4}n}} $
  $$= (1)1 + (-1)1 + (2)1 + 0 = 1 - 1 + 2 = 2 + 0j $$ 
  \\
  Temos assim:
  \begin{center}
  $
  \begin{bmatrix}
      -1+j & -1-j \\ 
      4 + 0j & 2 + 0j \\
  \end{bmatrix}
  $
  \end{center}
